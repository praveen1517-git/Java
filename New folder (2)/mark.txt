CREATE OR REPLACE PROCEDURE display_student_result (
    mark1 IN NUMBER, 
    mark2 IN NUMBER, 
    mark3 IN NUMBER
) IS
    total_marks NUMBER;
    result_status VARCHAR2(10);
BEGIN
    -- Call the function to get total marks
    total_marks := calculate_total_marks(mark1, mark2, mark3);
    
    -- Condition: If total marks ≥ 150 → PASS Else → FAIL
    IF total_marks >= 150 THEN
        result_status := 'PASS';
    ELSE
        result_status := 'FAIL';
    END IF;
    
    -- Display results using the DBMS_OUTPUT package
    DBMS_OUTPUT.PUT_LINE('Total Marks: ' || total_marks);
    DBMS_OUTPUT.PUT_LINE('Result: ' || result_status);
END;
/

CREATE OR REPLACE FUNCTION calculate_total_marks (
    mark1 IN NUMBER, 
    mark2 IN NUMBER, 
    mark3 IN NUMBER
) RETURN NUMBER IS
    total NUMBER;
BEGIN
    total := mark1 + mark2 + mark3;
    RETURN total;
END;
/
